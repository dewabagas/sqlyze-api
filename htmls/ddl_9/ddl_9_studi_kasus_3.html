Database Sistem Ecommerce
  1. Membuat Database
    2. CREATE DATABASE ecommerce;
  2. Membuat Tabel Produk
    3. CREATE TABLE produk (
id INT PRIMARY KEY AUTO_INCREMENT,
nama_produk VARCHAR(50) NOT NULL,
harga DECIMAL(10,2) NOT NULL,
kategori_id INT);
  3. Membuat Tabel Kategori
    4. CREATE TABLE kategori (
id INT PRIMARY KEY AUTO_INCREMENT,
nama_kategori VARCHAR(50) NOT NULL);
  4. Membuat Tabel Pelanggan
    5. CREATE TABLE pelanggan (
id INT PRIMARY KEY AUTO_INCREMENT,
nama VARCHAR(50) NOT NULL,
alamat TEXT,
email VARCHAR(50) UNIQUE NOT NULL);
  5. Membuat Tabel Mata Pesanan
    6. CREATE TABLE pesanan (
id INT PRIMARY KEY AUTO_INCREMENT,
pelanggan_id INT,
tanggal_pemesanan DATE NOT NULL,
status ENUM('Dikemas', 'Dikirim', 'Diterima', 'Batal') NOT NULL);
