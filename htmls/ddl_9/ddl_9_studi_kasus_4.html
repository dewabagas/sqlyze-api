Database Sistem Manajemen Inventori
  1. Membuat Database
    2. CREATE DATABASE inventori;
  2. Membuat Tabel Barang
    3. CREATE TABLE barang (
id INT PRIMARY KEY AUTO_INCREMENT,
nama_barang VARCHAR(50) NOT NULL,
jumlah INT NOT NULL,
harga_beli DECIMAL(10,2) NOT NULL,
harga_jual DECIMAL(10,2) NOT NULL);
  3. Membuat Tabel Supplier
    4. CREATE TABLE supplier (
id INT PRIMARY KEY AUTO_INCREMENT,
nama_supplier VARCHAR(50) NOT NULL,
alamat TEXT,
telepon VARCHAR(20));
  4. Membuat Tabel Transaksi
    5. CREATE TABLE transaksi (
id INT PRIMARY KEY AUTO_INCREMENT,
barang_id INT,
supplier_id INT,
tanggal_transaksi DATE NOT NULL,
tipe ENUM('Masuk', 'Keluar') NOT NULL,
jumlah INT NOT NULL);
